Program 1: Basic Calculator
 Q: Write a program to create a basic calculator for addition, subtraction, multiplication, 
 and division. The program should ask for two floating-point numbers and perform all four operations.
 
 
 import java.util.Scanner;

class BasicCalculator {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // Take user input
        System.out.print("Enter first number: ");
        double number1 = input.nextDouble();

        System.out.print("Enter second number: ");
        double number2 = input.nextDouble();

        // Perform operations
        double sum = number1 + number2;
        double difference = number1 - number2;
        double product = number1 * number2;
        double quotient = number1 / number2;

        // Display result
        System.out.println("The addition, subtraction, multiplication, and division value of 2 numbers " +
                           number1 + " and " + number2 + " is " +
                           sum + ", " + difference + ", " + product + ", and " + quotient);
    }
}

        // Calculate area in cmÂ² and convert to inÂ²

 Program 4: Convert Feet to Yards and Miles
ðŸ“˜ Q: Write a program to find the distance in yards and miles for the distance provided by the user in feet.
ðŸ“Œ Hint: 1 yard = 3 feet, 1 mile = 1760 yards


import java.util.Scanner;

class FeetToYardsAndMiles {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // Input
        System.out.print("Enter distance in feet: ");
        double distanceFeet = input.nextDouble();

        // Conversion
        double distanceYards = distanceFeet / 3;
        double distanceMiles = distanceYards / 1760;

        // Output
        System.out.println("The distance in yards is " + distanceYards + 
                           " while the distance in miles is " + distanceMiles);
    }
}
